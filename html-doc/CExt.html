val tagger_split_lines : string -> string list

Split lines using linefeed ('\n') as delimiter. For instance, tagger_split_lines "foo\nbar" returns ["foo"; "bar"].

val tagger_split_tabs : string -> string list

Splits text using tabulation ('\t') as delimiter. For instance, tagger_split_tabs "foo\tbar" returns ["foo"; "bar"].
